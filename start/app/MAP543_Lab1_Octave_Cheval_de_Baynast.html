def linear_reg(X_train, Y_train):
    w = np.linalg.solve((X_train.T @ X_train), (X_train.T @ Y_train))
    y_pred = np.dot(w, X_train.T)
    return y_pred

def MSE_calc(Y_true, Y_pred):
    MSE = np.square(np.subtract(Y_true, Y_pred)).mean()
    return MSE

def basis_poly(x):
    j = np.arange(1, 6)
    x = (x ** j)
    v = np.ones(len(x))
    return np.vstack((v, x.T)).T

def basis_exp(x):
    u = np.arange(0, 50, 5)
    x = np.exp( - ((40*x - u) ** 2) / 25)
    v = np.ones(len(x))
    return np.vstack((v, x.T)).T

def basis_cos1(x):
    j = np.arange(1, 6)
    x = np.cos(x / j)
    v = np.ones(len(x))
    return np.vstack((v, x.T)).T

def basis_cos2(x):
    j = np.arange(1, 26)
    x = np.cos(x / j)
    v = np.ones(len(x))
    return np.vstack((v, x.T)).T

# To get smooth regression curves I define a new linear regression function
# This can accomodate a low-interval variable that will span across the range of X

def linear_reg2(X_train, Y_train, x_linsp):
    w = np.linalg.solve((X_train.T @ X_train), (X_train.T @ Y_train))
    y_pred = np.dot(w, X_train.T)
    y_linsp = np.dot(w, x_linsp.T)
    return y_pred, y_linsp

The average MSE for the cos1 regression is  22.658838283869297 with a spread of  7.573243553048421
The average MSE for the cos2 regression is  22.667288233797024 with a spread of  7.590247366750685

These scores indicate that the $cos1$ regression function is the most generalizable. This is as it has both: the lowest average value for MSE, and the smallest spread between the two MSE values found.

Even in the case that one were to disregard all the regression results for regressions relating the number of sunspots and the number of Republicans in the Senate for a given year, the result would remain the same. One might want to do this as the connection between sunspots and Republicans is not easily logically made.
